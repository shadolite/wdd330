  • Stack traces are the best!
  • JavaScript sometimes fails silently in the background
  • Strict mode encourages better quality JavaScript
    - Can be used per function
    - Modules are strict by default
  • ES Lint tests code for quality
    - Can prevent commits containing errors
    - Requires npm
  • Use failure detection and exception handling
  • Jest can be used for test driven development
    - Requires npm 
    function squareRoot(number) {
  'use strict';
  if (number < 0) {
    throw new RangeError("You can't find the square root of negative numbers")
  }
  return Math.sqrt(number);
};

test('square root of 4 is 2', () => {
  expect(squareRoot(4)).toBe(2);
});
